### 今日新建/修改笔记

```dataviewjs
// 从当前日报的 frontmatter date 属性提取日期
const currentDate = dv.current().date;

let targetDate;
if (currentDate) {
  targetDate = String(currentDate).substring(0, 10);
} else {
  // 模板文件没有 date 属性时，使用今天日期
  targetDate = new Date().toISOString().substring(0, 10);
}

const pages = dv.pages()
  .where(p => {
    if (p.file.folder.includes("0-System")) return false;
    if (p.file.folder.includes("2-Periodic Notes")) return false;
    if (p.file.folder.includes("ibooks-highlights")) return false;
    if (p.file.folder.includes("Omnivore")) return false;

    // 检查笔记的 date 或 modified 属性是否匹配目标日期
    const noteDate = p.date ? String(p.date).substring(0, 10) : null;
    const noteModified = p.modified ? String(p.modified).substring(0, 10) : null;

    return noteDate === targetDate || noteModified === targetDate;
  })
  .sort(p => p.modified || p.date, 'desc')
  .slice(0, 15);

if (pages.length > 0) {
  dv.table(
    ["📝 笔记", "📁 目录", "📅 日期", "✏️ 修改"],
    pages.map(p => [
      p.file.link,
      p.file.folder,
      p.date ? String(p.date).substring(0, 10) : "-",
      p.modified ? String(p.modified).substring(0, 10) : "-"
    ])
  );
} else {
  dv.paragraph("当日暂无新建/修改笔记");
}
```
